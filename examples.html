Пример полного проекта


page = html(
    head(
        title("Мой проект"),
        link(href="styles.css"),
        script(src="app.js")
    ),
    body(
        h1("Добро пожаловать!", style="color:green;"),
        card("Карточка 1", "Описание 1", "image1.jpg"),
        card("Карточка 2", "Описание 2", "image2.jpg")
    )
)
save_html("index.html", page)
